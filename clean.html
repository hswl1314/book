#第二步----------------------------------------------------------------------------------
library(readr)
data <- read.csv("matrix.csv")
new_col_names <- c("c1","c2","c3","k1", "k2","k3")
colnames(data)[2:7] <- new_col_names #看有多少个样品
cleaned_data <- data[data$GeneSymbol != "", ]
cleaned_data <- na.omit(cleaned_data)
data_unique <- cleaned_data[!duplicated(cleaned_data[[1]]), ]
write.csv(data_unique, "cleaned_matrix.csv")
